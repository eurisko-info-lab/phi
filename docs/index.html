-- A streaming JSON parser in Phi

data JsonValue where
  JNull   : JsonValue
  JBool   : Bool -> JsonValue
  JNumber : Double -> JsonValue
  JString : String -> JsonValue
  JArray  : List JsonValue -> JsonValue
  JObject : Map String JsonValue -> JsonValue

-- Parser is a Cofree over the parsing functor
type Parser a = Cofree ParserF a

-- Compile to different targets
compile : Parser JsonValue -> Target -> Code
compile p WASM   = toWasm (optimize p)
compile p Native = toLLVM (optimize p)
compile p JVM    = toScala (optimize p)
